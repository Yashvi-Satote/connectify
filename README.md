    <ul>
        <li>➕ Add New Contact</li>
        <li>❌ Delete Contact</li>
        <li>✏️ Update Contact</li>
        <li>🔍 Search by Name</li>
        <li>📱 Search by Phone Number</li>
        <li>📋 View All Contacts</li>
        <li>🔤 Sort Contacts Alphabetically</li>
        <li>💾 Save Contacts to File</li>
        <li>📂 Load Contacts from File</li>
    </ul>
